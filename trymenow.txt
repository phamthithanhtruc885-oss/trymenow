        // --- 3. TOAST & ALERTS (Never use window.alert!) ---
        function showToast(message) {
            const toast = document.getElementById("toast");
            const toastMsg = document.getElementById("toast-message");
            toastMsg.innerText = message;
            
            toast.classList.remove("translate-y-20", "opacity-0");
            toast.classList.add("translate-y-0", "opacity-100");
            
            setTimeout(() => {
                toast.classList.remove("translate-y-0", "opacity-100");
                toast.classList.add("translate-y-20", "opacity-0");
            }, 3000);
        }

        // --- 4. NAVIGATION TAB SYSTEM ---
        function navigateTo(targetPageId) {
            // Hide all pages
            const sections = document.querySelectorAll('.page-section');
            sections.forEach(sec => sec.classList.remove('active'));

            // Show active page
            const targetSection = document.getElementById(`page-${targetPageId}`);
            if (targetSection) {
                targetSection.classList.add('active');
            }

            // Update Header links
            const navLinks = document.querySelectorAll('.nav-link');
            navLinks.forEach(link => {
                link.classList.remove('text-gold');
                link.classList.add('text-gray-300');
            });

            // Handle scroll up to top smoothly
            window.scrollTo({ top: 0, behavior: 'smooth' });

            // Re-render specifically for Cart page to reflect state changes
            if (targetPageId === 'cart') {
                renderCart();
            }
        }

        function setupNavigationActiveStates() {
            const navLinks = document.querySelectorAll('.nav-link');
            navLinks.forEach((link, idx) => {
                link.addEventListener('click', function() {
                    navLinks.forEach(l => {
                        l.classList.remove('text-gold');
                        l.classList.add('text-gray-300');
                    });
                    this.classList.remove('text-gray-300');
                    this.classList.add('text-gold');
                });
            });
        }

        function toggleMobileMenu() {
            const menu = document.getElementById("mobile-menu");
            menu.classList.toggle("hidden");
        }

        function renderShopProducts() {
            const grid = document.getElementById("shop-products-grid");
            const emptyMsg = document.getElementById("empty-shop-msg");
            if (!grid) return;

            // 1. FILTERING
            let filtered = PRODUCTS.filter(prod => {
                // Category match
                if (selectedCategoryFilter !== "all" && prod.category !== selectedCategoryFilter) {
                    return false;
                }
                
                // Price match
                if (selectedPriceFilter !== "all") {
                    if (selectedPriceFilter === "0-200" && prod.price > 200000) return false;
                    if (selectedPriceFilter === "200-400" && (prod.price < 200000 || prod.price > 400000)) return false;
                    if (selectedPriceFilter === "400+" && prod.price < 400000) return false;
                }

                // Size match
                if (selectedSizesFilter.length > 0) {
                    // check if product contains any of the selected sizes
                    const hasSize = prod.sizes.some(size => selectedSizesFilter.includes(size));
                    if (!hasSize) return false;
                }

                return true;
            });

            // 2. SORTING
            const sortVal = document.getElementById("sort-filter").value;
            if (sortVal === "price-asc") {
                filtered.sort((a, b) => a.price - b.price);
            } else if (sortVal === "price-desc") {
                filtered.sort((a, b) => b.price - a.price);
            }

            // 3. RENDER
            grid.innerHTML = "";
            document.getElementById("products-count").innerText = filtered.length;

            if (filtered.length === 0) {
                grid.classList.add("hidden");
                emptyMsg.classList.remove("hidden");
            } else {
                grid.classList.remove("hidden");
                emptyMsg.classList.add("hidden");

                filtered.forEach(prod => {
                    grid.innerHTML += createProductCardHTML(prod);
                });
            }
        }

        // --- 6. FILTER INTERACTIONS ---
        function filterByCategory(cat) {
            // Uncheck other radios
            const radios = document.getElementsByName("category-filter");
            radios.forEach(r => {
                if (r.value === cat) r.checked = true;
                else r.checked = false;
            });
            selectedCategoryFilter = cat;
            navigateTo("shop");
            renderShopProducts();
        }

        function filterProducts() {
            // Get Category Radio
            const catRadios = document.getElementsByName("category-filter");
            catRadios.forEach(r => {
                if (r.checked) selectedCategoryFilter = r.value;
            });

            // Get Price Radio
            const priceRadios = document.getElementsByName("price-filter");
            priceRadios.forEach(p => {
                if (p.checked) selectedPriceFilter = p.value;
            });

            renderShopProducts();
        }

        function toggleSizeFilter(size) {
            const idx = selectedSizesFilter.indexOf(size);
            if (idx > -1) {
                selectedSizesFilter.splice(idx, 1);
            } else {
                selectedSizesFilter.push(size);
            }

            // Update UI Active status of size buttons in filter sidebar
            const btns = document.querySelectorAll(".size-btn");
            btns.forEach(btn => {
                if (selectedSizesFilter.includes(btn.innerText)) {
                    btn.classList.add("bg-black", "text-white", "border-black");
                    btn.classList.remove("border-gray-200");
                } else {
                    btn.classList.remove("bg-black", "text-white", "border-black");
                    btn.classList.add("border-gray-200");
                }
            });

            renderShopProducts();
        }

        function resetFilters() {
            selectedCategoryFilter = "all";
            selectedPriceFilter = "all";
            selectedSizesFilter = [];
            
            // reset radio buttons
            document.getElementsByName("category-filter").forEach(r => r.value === "all" ? r.checked = true : r.checked = false);
            document.getElementsByName("price-filter").forEach(p => p.value === "all" ? p.checked = true : p.checked = false);
            
            // reset size buttons active visual
            const btns = document.querySelectorAll(".size-btn");
            btns.forEach(btn => {
                btn.classList.remove("bg-black", "text-white", "border-black");
                btn.classList.add("border-gray-200");
            });

            document.getElementById("sort-filter").value = "default";
            renderShopProducts();
        }

        // --- 7. QUICK VIEW MODAL INTERACTIONS ---
        function openQuickView(productId) {
            const product = PRODUCTS.find(p => p.id === productId);
            if (!product) return;

            activeQvProduct = product;
            selectedQvSize = product.sizes[0];
            selectedQvColor = product.colors[0];
            qvQuantity = 1;

            // Populate Modal Content
            document.getElementById("qv-name").innerText = product.name;
            document.getElementById("qv-image").src = product.image;
            document.getElementById("qv-price").innerText = formatPrice(product.price);
            document.getElementById("qv-original-price").innerText = formatPrice(product.originalPrice);
            document.getElementById("qv-desc").innerText = product.desc;
            document.getElementById("qv-badge").innerText = product.badge ? product.badge : "COLLECTION";
            document.getElementById("qv-quantity").innerText = qvQuantity;

            // Populate Sizes
            const sizesContainer = document.getElementById("qv-sizes-container");
            sizesContainer.innerHTML = "";
            product.sizes.forEach(size => {
                const activeClass = size === selectedQvSize ? "bg-black text-white border-black" : "border-gray-200 text-gray-700 hover:border-black";
                sizesContainer.innerHTML += `
                    <button onclick="selectQvSize('${size}')" id="qv-size-${size}" class="size-select-btn border text-xs px-3 py-2 font-bold rounded transition ${activeClass}">${size}</button>
                `;
            });

            // Populate Colors
            const colorsContainer = document.getElementById("qv-colors-container");
            colorsContainer.innerHTML = "";
            product.colors.forEach(color => {
                const activeClass = color === selectedQvColor ? "bg-black text-white border-black" : "border-gray-200 text-gray-700 hover:border-black";
                colorsContainer.innerHTML += `
                    <button onclick="selectQvColor('${color}')" id="qv-color-${color}" class="color-select-btn border text-xs px-3 py-2 font-bold rounded transition ${activeClass}">${color}</button>
                `;
            });

            // Show Modal
            document.getElementById("quick-view-modal").classList.remove("hidden");
        }

        // --- 8. CART & ORDER STATE LOGIC ---
        function quickAddToCart(productId) {
            const product = PRODUCTS.find(p => p.id === productId);
            if (!product) return;

            // Simple direct add with default size & color
            addToCart(product, product.sizes[0], product.colors[0], 1);
        }

        function addQvProductToCart() {
            if (!activeQvProduct) return;
            addToCart(activeQvProduct, selectedQvSize, selectedQvColor, qvQuantity);
            closeQuickView();
        }

        function addToCart(product, size, color, qty) {
            // Check if item already exists in cart with same size & color
            const existingIndex = cart.findIndex(item => item.product.id === product.id && item.size === size && item.color === color);

            if (existingIndex > -1) {
                cart[existingIndex].qty += qty;
            } else {
                cart.push({
                    product: product,
                    size: size,
                    color: color,
                    qty: qty
                });
            }

            updateCartBadge();
            saveCartToStorage();
            showToast(`Đã thêm ${qty} x ${product.name} vào giỏ hàng!`);
        }

        function updateCartBadge() {
            const badge = document.getElementById("cart-badge");
            const totalQty = cart.reduce((total, item) => total + item.qty, 0);
            badge.innerText = totalQty;
        }

        function saveCartToStorage() {
            localStorage.setItem("try_me_now_cart", JSON.stringify(cart));
        }

        function renderCart() {
            const itemsWrapper = document.getElementById("cart-items-wrapper");
            const container = document.getElementById("cart-container");
            const emptyView = document.getElementById("cart-empty-view");

            if (cart.length === 0) {
                container.classList.add("hidden");
                emptyView.classList.remove("hidden");
                return;
            }

            container.classList.remove("hidden");
            emptyView.classList.add("hidden");

            itemsWrapper.innerHTML = "";
            let subtotal = 0;

            cart.forEach((item, index) => {
                const itemTotal = item.product.price * item.qty;
                subtotal += itemTotal;

                itemsWrapper.innerHTML += `
                    <div class="flex items-center space-x-4 bg-white p-4 rounded-lg border border-gray-100 shadow-sm">
                        <!-- Product thumbnail -->
                        <div class="w-20 h-24 bg-gray-50 rounded overflow-hidden flex-shrink-0 cursor-pointer" onclick="openQuickView(${item.product.id})">
                            <img src="${item.product.image}" alt="${item.product.name}" class="w-full h-full object-cover">
                        </div>
                        
                        <!-- Product text details -->
                        <div class="flex-grow min-w-0">
                            <h4 class="font-title font-bold text-sm text-black uppercase truncate cursor-pointer hover:text-gold" onclick="openQuickView(${item.product.id})">
                                ${item.product.name}
                            </h4>
                            <p class="text-xs text-gray-500 mt-1">Size: <span class="font-semibold text-black">${item.size}</span> | Màu: <span class="font-semibold text-black">${item.color}</span></p>
                            <span class="text-xs font-bold text-black mt-2 block">${formatPrice(item.product.price)}</span>
                        </div>

                        <!-- Quantity control -->
                        <div class="flex flex-col items-end justify-between h-20">
                            <!-- Delete button -->
                            <button onclick="removeFromCart(${index})" class="text-gray-400 hover:text-red-500 transition duration-150 p-1" title="Xóa">
                                <i class="far fa-trash-alt"></i>
                            </button>

                            <!-- Quantity Selector -->
                            <div class="flex border border-gray-200 rounded items-center">
                                <button onclick="adjustCartQuantity(${index}, -1)" class="px-2 py-1 text-gray-500 hover:text-black"><i class="fas fa-minus text-[10px]"></i></button>
                                <span class="px-3 py-1 font-bold text-xs text-black">${item.qty}</span>
                                <button onclick="adjustCartQuantity(${index}, 1)" class="px-2 py-1 text-gray-500 hover:text-black"><i class="fas fa-plus text-[10px]"></i></button>
                            </div>
                        </div>
                    </div>
                `;
            });

            // Update Billing Pricing Info
            document.getElementById("cart-subtotal").innerText = formatPrice(subtotal);
            
            // Shipping calculation (free over 500k)
            const shippingCost = subtotal >= 500000 ? 0 : 30000;
            document.getElementById("cart-shipping").innerText = shippingCost === 0 ? "Miễn phí" : formatPrice(shippingCost);

            // Discount code logic
            const total = subtotal + shippingCost - currentDiscount;
            document.getElementById("cart-discount").innerText = `-${formatPrice(currentDiscount)}`;
            document.getElementById("cart-total").innerText = formatPrice(total < 0 ? 0 : total);
        }

        function adjustCartQuantity(index, amount) {
            cart[index].qty += amount;
            if (cart[index].qty < 1) cart[index].qty = 1;
            saveCartToStorage();
            updateCartBadge();
            renderCart();
        }

        function removeFromCart(index) {
            const name = cart[index].product.name;
            cart.splice(index, 1);
            saveCartToStorage();
            updateCartBadge();
            renderCart();
            showToast(`Đã xóa ${name} khỏi giỏ hàng.`);
        }

        function applyCoupon() {
            const val = document.getElementById("coupon-input").value.trim().toUpperCase();
            const msg = document.getElementById("coupon-message");
            
            if (val === "TRYME50") {
                currentDiscount = 50000;
                couponCodeApplied = val;
                msg.innerText = "Chúc mừng! Áp dụng thành công mã giảm 50.000₫.";
                msg.className = "text-xs mt-2 font-medium text-green-600 block";
                renderCart();
            } else if (val === "GENZ20") {
                currentDiscount = 20000;
                couponCodeApplied = val;
                msg.innerText = "Chúc mừng! Áp dụng thành công mã giảm 20.000₫.";
                msg.className = "text-xs mt-2 font-medium text-green-600 block";
                renderCart();
            } else if (val === "") {
                msg.innerText = "Bạn chưa nhập mã giảm giá.";
                msg.className = "text-xs mt-2 font-medium text-red-500 block";
            } else {
                msg.innerText = "Mã voucher không tồn tại hoặc đã hết lượt dùng.";
                msg.className = "text-xs mt-2 font-medium text-red-500 block";
            }
        }

        // --- 12. UTILITY FUNCTIONS ---
        function formatPrice(number) {
            return new Intl.NumberFormat('vi-VN', { style: 'currency', currency: 'VND' }).format(number);
        }

        function handleSearch(event) {
            if (event.key === "Enter") {
                triggerSearch(event.target.id);
            }
        }

        function triggerSearch(inputId) {
            const query = document.getElementById(inputId).value.trim().toLowerCase();
            if (!query) {
                showToast("Vui lòng nhập từ khóa tìm kiếm.");
                return;
            }

            // Redirect to shop page and filter
            navigateTo("shop");
            
            // Uncheck all radios in shop sidebar to allow full search scope
            resetFilters();

            // Custom search logic inside products list
            const grid = document.getElementById("shop-products-grid");
            const emptyMsg = document.getElementById("empty-shop-msg");

            let filtered = PRODUCTS.filter(prod => {
                return prod.name.toLowerCase().includes(query) || prod.category.toLowerCase().includes(query);
            });

            grid.innerHTML = "";
            document.getElementById("products-count").innerText = filtered.length;

            if (filtered.length === 0) {
                grid.classList.add("hidden");
                emptyMsg.classList.remove("hidden");
            } else {
                grid.classList.remove("hidden");
                emptyMsg.classList.add("hidden");

                filtered.forEach(prod => {
                    grid.innerHTML += createProductCardHTML(prod);
                });
            }

            document.getElementById("desktop-search").value = "";
            document.getElementById("mobile-search").value = "";
